CREATE TABLE Instituicao (
    ID int NOT NULL AUTO_INCREMENT,
    Nome varchar(255),
    Telefone varchar(255),
    Email varchar(255),
    CONSTRAINT PK_Instituicao PRIMARY KEY (ID)
)

CREATE TABLE Responsavel (
    ID int NOT NULL AUTO_INCREMENT,
    Nome varchar(255),
    Telefone varchar(255),
    Email varchar(255),
    Instituicao int,
    CONSTRAINT PK_Responsavel PRIMARY KEY (ID),
    FOREIGN KEY (Instituicao) REFERENCES Instituicao (ID)
)

CREATE TABLE Evento (
    ID int NOT NULL AUTO_INCREMENT,
    Nome varchar(255),
    Endereco varchar(255),
    Data date,
    HoraInicio time,
    HoraTermino time,
    Site varchar(255),
    Objetivo text,
    TempoApresentacao varchar(255),
    PassagemDeSom time,
    Cronograma text,
    Instituicao int,
    Responsavel int,
    CONSTRAINT PK_Evento PRIMARY KEY (ID),
    FOREIGN KEY (Instituicao) REFERENCES Instituicao (ID),
    FOREIGN KEY (Responsavel) REFERENCES Responsavel (ID)
)

CREATE TABLE Estrutura (
    ID int NOT NULL AUTO_INCREMENT,
    Microfones int,
    Pedestais int,
    Outros text,
    Observacoes text,
    Evento int,
    CONSTRAINT PK_Estrutura PRIMARY KEY (ID),
    FOREIGN KEY (Evento) REFERENCES Evento (ID)
)
